1. Exercise 1

  • Write a SARL program to sum all the items in a list.

Answer

Two answers are possible. Answer #1 is: 

import java.util.List
class Solution1 {
	static def sum(list : List) : Number {
		var s = 0.0
		for (e : list) {
			s += e
		}
		return s
	}
}
Answer #2 is: 

import java.util.List
class Solution2 {
	static def sum(list : List) : Number {
		list.reduce[accumulator, current | accumulator.doubleValue + current.doubleValue]
	}
}

## 3. Exercise 3 * Write a SARL program to get the largest number from a list. Answer
Two answers are possible. Answer #1 is: 

import java.util.List	
class Solution1 {
	static def maxValue(list : List) : Number {
		var m : Number = null
		for (e : list) {
			if (m === null || m < e) {
				m = e
			}
		}
		return m
	}
}
Answer #2 is: 

import java.util.List
class Solution2 {
	static def maxValue(list : List) : Number {
		list.max[a, b | a <=> b]
	}
}
## 4. Exercise 4 * Write a SARL program to count the number of strings from a given list of strings. The string length is 2 or more and the first and last characters are the same. * Sample List : `['abc', 'xyz', 'aba', '1221']` * Expected Result : `2` Answer
Two answers are possible. Answer #1 is: 

import java.util.List	
class Solution1 {
	static var original = #['abc', 'xyz', 'aba', '1221']
	static def countStrings(list : List) : int {
		var n = 0
		for (s : list) {
			if (s.length >= 2 && s.charAt(0) == s.charAt(s.length - 1)) {
				n++
			}
		}
		return n
	}
	static def main : void {
		println(countStrings(original))
	}
}
Answer #2 is: 

import java.util.List
class Solution2 {
	static var original = #['abc', 'xyz', 'aba', '1221']
	static def countStrings(list : List) : int {
		list.filter[it.length >= 2 && it.charAt(0) == it.charAt(it.length - 1)].size
	}
	static def main : void {
		println(countStrings(original))
	}
}
## 5. Exercise 5 * Write a SARL program to get a list, sorted in increasing order by the last element in each tuple from a given list of non-empty tuples. * Sample List : `[(2, 5), (1, 2), (4, 4), (2, 3), (2, 1)]` * Expected Result : `[(2, 1), (1, 2), (2, 3), (4, 4), (2, 5)]` Answer 

import java.util.List	
class Solution {
	static var original = #[#[2, 5], #[1, 2], #[4, 4], #[2, 3], #[2, 1]]
	static def main : void {
		var result = original.sortWith[a, b | 
			a.get(1) <=> b.get(1)
		]
		println(result)
	}
}

import java.util.List	
class Solution {
	static def remove_duplicates(list : List) {
		var result = newArrayList
		for (e : list) {
			if (!result.contains(e)) {
				result += e
			}
		}
		return result
	}
}

## 10. Exercise 10

* Write a SARL function that takes two lists and returns `true` if they have at least one common member.

Answer


Two answers are possible. Answer #1 is:


import java.util.List	
class Solution1 {
	static def common(list1 : List, list2 : List) {
		for (e : list1) {
			if (list2.contains(e)) {
				return true
			}
		}
		return false
	}
}



Answer #2 is:


import java.util.List
class Solution2 {
	static def words(list1 : List, list2 : List) {
		list1.exists[list2.contains(it)]
	}
}

## 12. Exercise 12

* Write a SARL program to generate a 3*4*6 3D array whose each element is `*`.

Answer


Two answers are possible. Answer #1 is:


class Solution1 {
	static def main : void {
		var result = newArrayList
		for (a : 1..3) {
			var dim2 = newArrayList
			for (b : 1..4) {
				var dim3 = newArrayList
				for (c : 1..6) {
					dim3 += '*'
				}
				dim2 += dim3
			}
			result += dim2
		}
		println(result)
	}
}



Answer #2 is:


import java.util.List
class Solution2 {
	static def main : void {
		var result = (1..3).map[
			(1..4).map[
				(1..6).map['*'].toList
			].toList
		].toList
		println(result)
	}
}
